function runTask1() {
  let val = prompt("Введіть число:");
  let num = Number(val);
  let status = "";

  if (val === null || val.trim() === "") {
      status = "Ви нічого не ввели";
  } else if (isNaN(num)) {
      status = "Це не число!";
  } else if (num > 0) {
      status = "Число додатнє";
  } else if (num < 0) {
      status = "Число від’ємне";
  } else {
      status = "Число дорівнює нулю";
  }

  alert(status);
  let msg = `Введено: "${val}", Результат: ${status}`;
  console.log(msg);
  document.getElementById('log1').innerText = msg;
}

function runTask2() {
  let val = prompt("Введіть число 1-4 (1-Зима, 2-Весна...):");
  let result = "";
  switch(val) {
      case '1': result = "Зима"; break;
      case '2': result = "Весна"; break;
      case '3': result = "Літо"; break;
      case '4': result = "Осінь"; break;
      default: result = "Невідома пора року";

  console.log(`Змінна result: ${result}`);
  document.getElementById('log2').innerText = `Введено: ${val}\nЗмінна result: ${result}\n(Див. також консоль)`;
}

Завдання №6: Фільтрація масивів

Програмний код

function filterArray(numbers, value) {
  let newArray = [];
  for (let i = 0; i < numbers.length; i++) {
      if (numbers[i] > value) {
          newArray.push(numbers[i]);
      }
  }
  return newArray;
}

function runTask6() {
    let arrStr = prompt("Введіть числа масиву через кому:", "1, 10, 20, 5, 30");
    if (!arrStr) return; // Скасовано

    let valStr = prompt("Введіть число для фільтрації:", "7");
    if (valStr === null) return;

    // Перевірка на коректність вводу
    if (valStr.trim() === "" || isNaN(valStr)) {
        alert("Помилка: некоректне значення фільтру!");
        return;
    }
    
    // Перевірка масиву
    let parts = arrStr.split(',');
    let numbers = [];
    for(let p of parts) {
        if(p.trim() === "" || isNaN(p)) {
             alert("Помилка: масив має містити тільки числа!");
             return;
        }
        numbers.push(Number(p));
    }

    let value = Number(valStr);
    let result = filterArray(numbers, value);
    
    let msg = `Вхідний масив: [${numbers}]\nЧисла більше ${value}: [${result}]`;
    console.log(msg);
    document.getElementById('log6').innerText = msg;
}

Завдання №7 (Варіант 7)

1. Знайти найбільший серед від’ємних та найменший серед додатних елементів масиву.
2. Упорядкувати масив у порядку зменшення методом вибору (Selection Sort).

Програмний код

function runTask7() {
    let sizeStr = prompt("Введіть кількість елементів масиву:", "10");
    if (sizeStr === null) return;
    
    let size = Number(sizeStr);
    if(isNaN(size) || size <= 0) {
        alert("Введіть коректне число > 0");
        return;
    }

    // Генерація
    let arr = Array.from({length: size}, () => Math.floor(Math.random() * 200) - 100);
    let output = `Вхідний масив: [${arr.join(', ')}]\n\n`;

    // 1. Пошук
    let maxNeg = -Infinity, minPos = Infinity;
    let hasNeg = false, hasPos = false;
    
    for(let n of arr) {
        if (n < 0) { hasNeg = true; if (n > maxNeg) maxNeg = n; }
        if (n > 0) { hasPos = true; if (n < minPos) minPos = n; }
    }
    output += `Макс. від'ємне: ${hasNeg ? maxNeg : 'немає'}\n`;
    output += `Мін. додатне: ${hasPos ? minPos : 'немає'}\n\n`;

    // 2. Сортування вибором (Selection Sort)
    let sorted = [...arr];
    for (let i = 0; i < sorted.length - 1; i++) {
        let maxIdx = i;
        for (let j = i + 1; j < sorted.length; j++) {
            if (sorted[j] > sorted[maxIdx]) maxIdx = j;
        }
        let temp = sorted[i]; 
        sorted[i] = sorted[maxIdx]; 
        sorted[maxIdx] = temp;
    }
    output += `Відсортований (за зменшенням): [${sorted.join(', ')}]`;
    
    document.getElementById('res7').innerText = output;
}

function runTask8() {
    // Генерація масиву (10 випадкових чисел)
    let arr = Array.from({length: 10}, () => Math.floor(Math.random() * 200) - 100);
    let pos = arr.filter(n => n >= 0);
    
    let logMsg = `Згенерований масив: [${arr}]\nДодатні числа: [${pos}]`;
    
    if(pos.length >= 3) {
       let newVal = prompt(`Третій елемент додатних (${pos[2]}). Введіть ВІД'ЄМНЕ число для заміни:`);
       
       if (newVal === null) {
           logMsg += `\nСкасовано.`;
       } else if (newVal.trim() === "" || isNaN(newVal)) {
           alert("Помилка: це не число!");
           logMsg += `\nПомилка введення.`;
       } else if (Number(newVal) >= 0) {
           alert("Помилка: число має бути від'ємним!");
           logMsg += `\nПомилка: введено не від'ємне число.`;
       } else {
           pos[2] = Number(newVal);
           logMsg += `\nУспішна заміна! Новий масив додатних: [${pos}]`;
       }
    } else {
       logMsg += `\nНедостатньо додатних елементів (< 3) для заміни.`;
    }
    document.getElementById('log8').innerText = logMsg;
}

Завдання №9: Гістограма (Варіант 7)

Програмний код

function buildHistogram() {
  const input = document.getElementById('histInput').value;
  const container = document.getElementById('chart');
  container.innerHTML = '';

  // Валідація: дозволені цифри, коми, мінуси, крапки, пробіли
  if (!/^[\d,\s.-]+$/.test(input) || input.trim() === "") {
      alert("Помилка: введіть коректні числа через кому!");
      return;
  }
  
  const nums = input.split(',').map(n => Number(n.trim()));
  
  if (nums.some(isNaN)) {
      alert("Помилка: одне зі значень не є числом!");
      return;
  }

  if (nums.length === 0) return;

  const maxVal = Math.max(...nums);

  nums.forEach(n => {
      let bar = document.createElement('div');
      bar.className = 'bar';
      // Розрахунок висоти (якщо макс > 0)
      let h = (maxVal > 0) ? (n / maxVal) * 100 : 0;
      if (h < 0) h = 0; 
      
      bar.style.height = h + '%';
      bar.innerHTML = `${n}`;
      container.appendChild(bar);
  });
}
